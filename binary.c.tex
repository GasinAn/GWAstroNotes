\chapter{双星系统}

\section{基本公式}

\def\M{\mathcal{M}}
\begin{equation}
    \M:=\mu^{3/5}M^{2/5}
\end{equation}
\begin{equation}
    h_+=\frac{4\nu \M}{D}[\pi \M F(t)]^{2/3}\frac{1+\cos^2\iota}{2}\cos\Phi(t)
\end{equation}
\begin{equation}
    h_\times=\frac{4\nu \M}{D}[\pi \M F(t)]^{2/3}\cos\iota\sin\Phi(t)
\end{equation}
\begin{equation}
    h=F_+h_++F_\times h_\times
\end{equation}
